C++ Program: Stack Using Linked List

#include <iostream>
using namespace std;

// Node structure for stack
struct Node {
    int data;
    Node* next;
};

int main() {
    Node* top = NULL; // Points to the top of the stack
    Node* temp;
    int choice, value;

    do {
        cout << "\n--- Stack Menu ---";
        cout << "\n1. Push";
        cout << "\n2. Pop";
        cout << "\n3. Display";
        cout << "\n4. Exit";
        cout << "\nEnter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1: // Push
                temp = new Node;
                cout << "Enter value to push: ";
                cin >> value;
                temp->data = value;
                temp->next = top;
                top = temp;
                cout << value << " pushed onto the stack.";
                break;

            case 2: // Pop
                if (top == NULL)
                    cout << "Stack Underflow! Nothing to pop.";
                else {
                    cout << "Popped element: " << top->data;
                    temp = top;
                    top = top->next;
                    delete temp;
                }
                break;

            case 3: // Display
                if (top == NULL)
                    cout << "Stack is empty.";
                else {
                    cout << "Stack elements (top to bottom): ";
                    temp = top;
                    while (temp != NULL) {
                        cout << temp->data << " ";
                        temp = temp->next;
                    }
                    cout << endl;
                }
                break;

            case 4:
                cout << "Exiting program.";
                break;

            default:
                cout << "Invalid choice! Please try again.";
        }
    } while (choice != 4);
    return 0;
}

C++ Program: Infix to Postfix Conversion Using Stack

#include <iostream>
#include <cstring>
using namespace std;

// Node structure for linked list (stack)
struct Node {
    char data;
    Node* next;
};

// Function to return operator precedence
int priority(char c) {
    if (c == '+' || c == '-')
        return 1;
    else if (c == '*' || c == '/')
        return 2;
    else if (c == '(')
        return 0; // Lowest priority
    else
        return -1; // For operands
}

int main() {
    char infix[100];
    Node* top = NULL; // Stack top pointer
    Node* temp;

    cout << "Enter infix expression: ";
    cin >> infix;

    cout << "Postfix expression is: ";

    int i = 0;
    while (infix[i] != '\0') {
        char ch = infix[i];

        // Operand â†’ print directly
        if ((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z') || isdigit(ch)) {
            cout << ch;
        }

        // Left parenthesis â†’ push to stack
        else if (ch == '(') {
            temp = new Node;
            temp->data = ch;
            temp->next = top;
            top = temp;
        }

        // Right parenthesis â†’ pop till '('
        else if (ch == ')') {
            while (top != NULL && top->data != '(') {
                cout << top->data;
                temp = top;
                top = top->next;
                delete temp;
            }
            if (top != NULL && top->data == '(') {
                temp = top;
                top = top->next;
                delete temp;
            }
        }

        // Operator â†’ compare precedence and push/pop accordingly
        else {
            while (top != NULL && priority(top->data) >= priority(ch)) {
                cout << top->data;
                temp = top;
                top = top->next;
                delete temp;
            }
            temp = new Node;
            temp->data = ch;
            temp->next = top;
            top = temp;
        }
        i++;
    }

    // Pop any remaining operators from the stack
    while (top != NULL) {
        cout << top->data;
        temp = top;
        top = top->next;
        delete temp;
    }

    return 0;
}

C++ Program: Infix to Prefix Conversion Using Stack

#include <iostream>
#include <cstring>
#include <cctype> // for isalnum
using namespace std;

struct Node {
    char data;
    Node* next;
};

int main() {
    char infix[100], result[100];
    int i = 0, k = 0;

    cout << "Enter infix expression: ";
    cin >> infix;

    // Step 1: Reverse infix and swap '(' and ')'
    int len = strlen(infix);
    for (i = 0; i < len / 2; i++) {
        char temp = infix[i];
        infix[i] = infix[len - i - 1];
        infix[len - i - 1] = temp;
    }

    for (i = 0; i < len; i++) {
        if (infix[i] == '(')
            infix[i] = ')';
        else if (infix[i] == ')')
            infix[i] = '(';
    }

    // Step 2: Initialize stack
    Node* top = NULL;

    // Step 3: Process each character of reversed infix
    i = 0;
    while (infix[i] != '\0') {
        char ch = infix[i];

        // If operand, add to result
        if (isalnum(ch)) {
            result[k++] = ch;
        }

        // If '(', push to stack
        else if (ch == '(') {
            Node* temp = new Node;
            temp->data = ch;
            temp->next = top;
            top = temp;
        }

        // If ')', pop and add to result until '(' is found
        else if (ch == ')') {
            while (top != NULL && top->data != '(') {
                result[k++] = top->data;
                Node* temp = top;
                top = top->next;
                delete temp;
            }
            if (top != NULL && top->data == '(') {
                Node* temp = top;
                top = top->next;
                delete temp;
            }
        }

        // If operator
        else {
            int p1;
            if (ch == '+' || ch == '-')
                p1 = 1;
            else
                p1 = 2;

            while (top != NULL) {
                char topOp = top->data;
                int p2;
                if (topOp == '+' || topOp == '-') p2 = 1;
                else if (topOp == '*' || topOp == '/') p2 = 2;
                else p2 = 0;

                if (p2 >= p1) {
                    result[k++] = top->data;
                    Node* temp = top;
                    top = top->next;
                    delete temp;
                } else {
                    break;
                }
            }

            Node* temp = new Node;
            temp->data = ch;
            temp->next = top;
            top = temp;
        }
        i++;
    }

    // Step 4: Pop remaining operators
    while (top != NULL) {
        result[k++] = top->data;
        Node* temp = top;
        top = top->next;
        delete temp;
    }

    result[k] = '\0';

    // Step 5: Display final prefix expression
    cout << "Prefix expression is: " << result << endl;

    return 0;
}
