C++ Program: Stack Using Linked List

#include <iostream>
using namespace std;

// Node structure for stack
struct Node {
    int data;
    Node* next;
};

// Stack ADT using linked list
class Stack {
private:
    Node* top; // Points to top of the stack

public:
    Stack() {
        top = NULL;
    }

    // Push operation
    void push(int value) {
        Node* newNode = new Node;
        if (!newNode) {
            cout << "Stack Overflow! Memory not allocated.\n";
            return;
        }
        newNode->data = value;
        newNode->next = top;
        top = newNode;
        cout << value << " pushed onto the stack.\n";
    }

    // Pop operation
    void pop() {
        if (isEmpty()) {
            cout << "Stack Underflow! Nothing to pop.\n";
            return;
        }
        Node* temp = top;
        cout << "Popped element: " << top->data << endl;
        top = top->next;
        delete temp;
    }

    // Peek operation (View top element)
    void peek() {
        if (isEmpty()) {
            cout << "Stack is empty.\n";
        } else {
            cout << "Top element: " << top->data << endl;
        }
    }

    // Check if stack is empty
    bool isEmpty() {
        return (top == NULL);
    }

    // Check if stack is full (memory allocation failure)
    bool isFull() {
        Node* temp = new (nothrow) Node;
        if (!temp)
            return true;
        delete temp;
        return false;
    }

    // Display all elements
    void display() {
        if (isEmpty()) {
            cout << "Stack is empty.\n";
            return;
        }
        cout << "Stack elements (top to bottom): ";
        Node* temp = top;
        while (temp != NULL) {
            cout << temp->data << " ";
            temp = temp->next;
        }
        cout << endl;
    }

    // Destructor to clean memory
    ~Stack() {
        while (!isEmpty()) {
            pop();
        }
    }
};

// Main function
int main() {
    Stack s;
    int choice, value;

    do {
        cout << "\n--- Stack Menu ---";
        cout << "\n1. Push";
        cout << "\n2. Pop";
        cout << "\n3. Peek";
        cout << "\n4. IsEmpty";
        cout << "\n5. IsFull";
        cout << "\n6. Display";
        cout << "\n7. Exit";
        cout << "\nEnter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1:
                cout << "Enter value to push: ";
                cin >> value;
                s.push(value);
                break;

            case 2:
                s.pop();
                break;

            case 3:
                s.peek();
                break;

            case 4:
                if (s.isEmpty())
                    cout << "Stack is empty.\n";
                else
                    cout << "Stack is not empty.\n";
                break;

            case 5:
                if (s.isFull())
                    cout << "Stack is full (memory not available).\n";
                else
                    cout << "Stack is not full.\n";
                break;

            case 6:
                s.display();
                break;

            case 7:
                cout << "Exiting program.\n";
                break;

            default:
                cout << "Invalid choice! Please try again.\n";
        }

    } while (choice != 7);

    return 0;
}

#include <iostream>
#include <cstring>
using namespace std;

// Node structure for linked list (stack)
struct Node {
    char data;
    Node* next;
};

// Function to return operator precedence
int priority(char c) {
    if (c == '+' || c == '-')
        return 1;
    else if (c == '*' || c == '/')
        return 2;
    else if (c == '(')
        return 0; // Lowest priority
    else
        return -1; // For operands
}

int main() {
    char infix[100];
    Node* top = NULL; // Stack top pointer
    Node* temp;

    cout << "Enter infix expression: ";
    cin >> infix;

    cout << "Postfix expression is: ";

    int i = 0;
    while (infix[i] != '\0') {
        char ch = infix[i];

        // Operand â†’ print directly
        if ((ch >= 'a' && ch <= 'z') || (ch >= 'A' && ch <= 'Z') || isdigit(ch)) {
            cout << ch;
        }

        // Left parenthesis â†’ push to stack
        else if (ch == '(') {
            temp = new Node;
            temp->data = ch;
            temp->next = top;
            top = temp;
        }

        // Right parenthesis â†’ pop till '('
        else if (ch == ')') {
            while (top != NULL && top->data != '(') {
                cout << top->data;
                temp = top;
                top = top->next;
                delete temp;
            }
            if (top != NULL && top->data == '(') {
                temp = top;
                top = top->next;
                delete temp;
            }
        }

        // Operator â†’ compare precedence and push/pop accordingly
        else {
            while (top != NULL && priority(top->data) >= priority(ch)) {
                cout << top->data;
                temp = top;
                top = top->next;
                delete temp;
            }
            temp = new Node;
            temp->data = ch;
            temp->next = top;
            top = temp;
        }
        i++;
    }

    // Pop any remaining operators from the stack
    while (top != NULL) {
        cout << top->data;
        temp = top;
        top = top->next;
        delete temp;
    }

    return 0;
}

C++ Program: Infix to Prefix Conversion Using Stack

#include <iostream>
#include <cstring>
#include <cctype> // for isalnum
using namespace std;

struct Node {
    char data;
    Node* next;
};

int main() {
    char infix[100], result[100];
    int i = 0, k = 0;

    cout << "Enter infix expression: ";
    cin >> infix;

    // Step 1: Reverse infix and swap '(' and ')'
    int len = strlen(infix);
    for (i = 0; i < len / 2; i++) {
        char temp = infix[i];
        infix[i] = infix[len - i - 1];
        infix[len - i - 1] = temp;
    }

    for (i = 0; i < len; i++) {
        if (infix[i] == '(')
            infix[i] = ')';
        else if (infix[i] == ')')
            infix[i] = '(';
    }

    // Step 2: Initialize stack
    Node* top = NULL;

    // Step 3: Process each character of reversed infix
    i = 0;
    while (infix[i] != '\0') {
        char ch = infix[i];

        // If operand, add to result
        if (isalnum(ch)) {
            result[k++] = ch;
        }

        // If '(', push to stack
        else if (ch == '(') {
            Node* temp = new Node;
            temp->data = ch;
            temp->next = top;
            top = temp;
        }

        // If ')', pop and add to result until '(' is found
        else if (ch == ')') {
            while (top != NULL && top->data != '(') {
                result[k++] = top->data;
                Node* temp = top;
                top = top->next;
                delete temp;
            }
            if (top != NULL && top->data == '(') {
                Node* temp = top;
                top = top->next;
                delete temp;
            }
        }

        // If operator
        else {
            int p1;
            if (ch == '+' || ch == '-')
                p1 = 1;
            else
                p1 = 2;

            while (top != NULL) {
                char topOp = top->data;
                int p2;
                if (topOp == '+' || topOp == '-') p2 = 1;
                else if (topOp == '*' || topOp == '/') p2 = 2;
                else p2 = 0;

                if (p2 >= p1) {
                    result[k++] = top->data;
                    Node* temp = top;
                    top = top->next;
                    delete temp;
                } else {
                    break;
                }
            }

            Node* temp = new Node;
            temp->data = ch;
            temp->next = top;
            top = temp;
        }
        i++;
    }

    // Step 4: Pop remaining operators
    while (top != NULL) {
        result[k++] = top->data;
        Node* temp = top;
        top = top->next;
        delete temp;
    }

    result[k] = '\0';

    // Step 5: Display final prefix expression
    cout << "Prefix expression is: " << result << endl;

    return 0;
}
