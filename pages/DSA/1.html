#include <iostream>
#include <vector>
#include <string>
using namespace std;

struct Student {
    int id;
    string name;
    float cgpa;
};

// Function to add student
void addStudent(vector<Student> &list) {
    Student s;
    cout << "\nEnter ID: ";
    cin >> s.id;
    cin.ignore();
    cout << "Enter Name: ";
    getline(cin, s.name);
    cout << "Enter CGPA: ";
    cin >> s.cgpa;
    list.push_back(s);
    cout << "Student added!\n";
}

// Function to display all students
void showStudents(const vector<Student> &list) {
    if (list.empty()) {
        cout << "No students found!\n";
        return;
    }
    cout << "\nID\tName\t\tCGPA\n";
    for (auto &s : list)
        cout << s.id << "\t" << s.name << "\t\t" << s.cgpa << "\n";
}

// Linear search by name
void linearSearch(const vector<Student> &list, string name) {
    bool found = false;
    for (auto &s : list) {
        if (s.name == name) {
            cout << "\nFound: ID=" << s.id << ", Name=" << s.name << ", CGPA=" << s.cgpa << "\n";
            found = true;
            break;
        }
    }
    if (!found)
        cout << "Student not found!\n";
}

// Sort students by name using Bubble Sort
void bubbleSortByName(vector<Student> &list) {
    int n = list.size();
    for (int i = 0; i < n - 1; i++) {
        for (int j = 0; j < n - i - 1; j++) {
            if (list[j].name > list[j + 1].name) {
                swap(list[j], list[j + 1]);
            }
        }
    }
    cout << "Students sorted by name using Bubble Sort!\n";
}

int main() {
    vector<Student> students;
    int choice;
    string name;

    while (true) {
        cout << "\n--- STUDENT DATABASE ---\n";
        cout << "1. Add Student\n2. Show Students\n3. Search by Name\n4. Sort by Name (Bubble Sort)\n5. Exit\n";
        cout << "Enter choice: ";
        cin >> choice;
        cin.ignore();

        if (choice == 1) addStudent(students);
        else if (choice == 2) showStudents(students);
        else if (choice == 3) {
            cout << "Enter name to search: ";
            getline(cin, name);
            linearSearch(students, name);
        }
        else if (choice == 4) bubbleSortByName(students);
        else if (choice == 5) {
            cout << "Exiting...\n";
            break;
        }
        else cout << "Invalid choice!\n";
    }
    return 0;
}

// Sort students by name using Selection Sort
void selectionSortByName(vector<Student> &list) {
    int n = list.size();
    for (int i = 0; i < n - 1; i++) {
        int minIndex = i;
        for (int j = i + 1; j < n; j++) {
            if (list[j].name < list[minIndex].name) {
                minIndex = j;
            }
        }
        if (minIndex != i) {
            swap(list[i], list[minIndex]);
        }
    }
    cout << "Students sorted by name using Selection Sort!\n";
}

#include <iostream>
#include <vector>
#include <string>
#include <algorithm>
using namespace std;

struct Student {
    int id;
    string name;
    float cgpa;
};

// Function to add student
void addStudent(vector<Student> &list) {
    Student s;
    cout << "\nEnter ID: ";
    cin >> s.id;
    cin.ignore();
    cout << "Enter Name: ";
    getline(cin, s.name);
    cout << "Enter CGPA: ";
    cin >> s.cgpa;
    list.push_back(s);
    cout << "Student added!\n";
}

// Function to display all students
void showStudents(const vector<Student> &list) {
    if (list.empty()) {
        cout << "No students found!\n";
        return;
    }
    cout << "\nID\tName\t\tCGPA\n";
    for (auto &s : list)
        cout << s.id << "\t" << s.name << "\t\t" << s.cgpa << "\n";
}

// Function to sort students by name (required for Binary Search)
void sortByName(vector<Student> &list) {
    sort(list.begin(), list.end(), [](Student a, Student b) {
        return a.name < b.name;
    });
}

// Binary Search by name
void binarySearch(const vector<Student> &list, string name) {
    int left = 0;
    int right = list.size() - 1;
    bool found = false;

    while (left <= right) {
        int mid = left + (right - left) / 2;
        if (list[mid].name == name) {
            cout << "\nFound: ID=" << list[mid].id << ", Name=" << list[mid].name << ", CGPA=" << list[mid].cgpa << "\n";
            found = true;
            break;
        } 
        else if (list[mid].name < name) {
            left = mid + 1;
        } else {
            right = mid - 1;
        }
    }

    if (!found) {
        cout << "Student not found!\n";
    }
}

int main() {
    vector<Student> students;
    int choice;
    string name;

    while (true) {
        cout << "\n--- STUDENT DATABASE ---\n";
        cout << "1. Add Student\n2. Show Students\n3. Search by Name (Binary Search)\n4. Exit\n";
        cout << "Enter choice: ";
        cin >> choice;

        else if (choice == 2) showStudents(students);
        else if (choice == 3) {
            if (students.empty()) {

                continue;

C++ Program: Bubble Sort

#include <iostream>
#include <stdlib.h>
#include <conio.h>
using namespace std;

int main() {
    int n, i, j, arr[10], temp;

    cout << "Enter Total Number of Elements: ";
    cin >> n;
     
    cout << "Enter Elements: ";
    for (i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Bubble Sort Algorithm
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    cout << "Sorted Elements in Ascending Order: ";
    for (i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}

C++ Program: Selection Sort

#include <iostream>
#include <stdlib.h>
#include <conio.h>

    int n, i, j, arr[10], temp;

    cout << "Enter Total Number of Elements: ";

        cin >> arr[i];
    }

    // Selection Sort Algorithm
    for (i = 0; i < n - 1; i++) {
        for (j = i + 1; j < n; j++) {
            if (arr[i] > arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }

    cout << "Sorted Elements: ";
    for (i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}
