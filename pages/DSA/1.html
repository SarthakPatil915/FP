C++ Program: Linear Search

#include <iostream>
using namespace std;

int main() {
    int n, a[10], i, flag = 0, Number;

    cout << "Enter Total Number of Elements: ";
    cin >> n;

    cout << "Enter Elements (Ascending Order): ";
    for (i = 0; i < n; i++) {
        cin >> a[i];
    }

    cout << "Enter Element to be Searched: ";
    cin >> Number;

    for (i = 0; i < n; i++) {
        if (Number == a[i]) {
            flag = 1;
            break;
        } else {
            flag = 0;
        }
    }

    if (flag == 1) {
        cout << "Found at Index: " << i << endl;
        cout << "Element Found at Position: " << i + 1;
    } else {
        cout << "Element Not Found";
    }

    return 0;
}

C++ Program: Binary Search

#include <iostream>
#include <stdlib.h>
#include <conio.h>
using namespace std;

int main() {
    int n, i, arr[10], number, low = 0, high, mid, flag = 0;

    cout << "Enter Total Number of Elements: ";
    cin >> n;

    cout << "Enter Elements (Ascending Order): ";
    for (i = 0; i < n; i++) {
        cin >> arr[i];
    }

    high = n - 1;
    cout << "Enter Element to be Searched: ";
    cin >> number;

    while (low <= high) {
        mid = (low + high) / 2;
        if (arr[mid] == number) {
            flag = 1;
            break;
        } else if (number > arr[mid]) {
            low = mid + 1;
        } else {
            high = mid - 1;
        }
    }

    if (flag == 1) {
        cout << "Found at Index: " << mid << endl;
        cout << "Element Found at Position: " << mid + 1;
    } else {
        cout << "Element Not Found";
    }

    return 0;
}

C++ Program: Bubble Sort

#include <iostream>
#include <stdlib.h>
#include <conio.h>
using namespace std;

int main() {
    int n, i, j, arr[10], temp;

    cout << "Enter Total Number of Elements: ";
    cin >> n;
     
    cout << "Enter Elements: ";
    for (i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Bubble Sort Algorithm
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    cout << "Sorted Elements in Ascending Order: ";
    for (i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}

C++ Program: Selection Sort

#include <iostream>
#include <stdlib.h>
#include <conio.h>
using namespace std;

int main() {
    int n, i, j, arr[10], temp;

    cout << "Enter Total Number of Elements: ";
    cin >> n;

    cout << "Enter Elements: ";
    for (i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Selection Sort Algorithm
    for (i = 0; i < n - 1; i++) {
        for (j = i + 1; j < n; j++) {
            if (arr[i] > arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }

    cout << "Sorted Elements: ";
    for (i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}

Enter Total Number of Elements: 5
Enter Elements: 40 10 30 20 50
Sorted Elements: 10 20 30 40 50
