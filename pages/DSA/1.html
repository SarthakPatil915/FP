Student Data Base (Bubble Sort)

#include <iostream>
#include <cstring>
using namespace std;

struct Student {
    int id;
    char name[50];
    float cgpa;
} s[10];

int count = 0;

// Add a student
void addStudent() {
    if (count >= 10) {
        cout << "Student limit reached!\n";
        return;
    }
    cout << "Enter Student ID: ";
    cin >> s[count].id;
    cin.ignore();
    cout << "Enter Student Name: ";
    cin.getline(s[count].name, 50);
    cout << "Enter CGPA: ";
    cin >> s[count].cgpa;
    count++;
}

// Display all students
void displayStudents() {
    if (count == 0) {
        cout << "No students to display.\n";
        return;
    }
    cout << "\n--- Student Records ---\n";
    for (int i = 0; i < count; i++)
        cout << "ID: " << s[i].id << ", Name: " << s[i].name
             << ", CGPA: " << s[i].cgpa << "\n";
}

// Bubble Sort by Name (Aâ€“Z)
void bubbleSortByName() {
    if (count == 0) {
        cout << "No students to sort.\n";
        return;
    }

    for (int i = 0; i < count - 1; i++) {
        for (int j = 0; j < count - i - 1; j++) {
            if (strcmp(s[j].name, s[j + 1].name) > 0) {
                Student temp = s[j];
                s[j] = s[j + 1];
                s[j + 1] = temp;
            }
        }
    }

    cout << "\nStudents sorted by Name (Aâ€“Z):\n";
    displayStudents();
}

// Main menu
int main() {
    int choice;
    do {
        cout << "\n=== Student Database (Bubble Sort) ===\n";
        cout << "1. Add Student\n2. Display Students\n3. Sort by Name (Aâ€“Z)\n4. Exit\n";
        cout << "Enter your choice: ";
        cin >> choice;

        switch (choice) {
            case 1: addStudent(); break;
            case 2: displayStudents(); break;
            case 3: bubbleSortByName(); break;
            case 4: cout << "Exiting...\n"; break;
            default: cout << "Invalid choice!\n";
        }
    } while (choice != 4);

    return 0;
}

#include <iostream>
#include <cstring>
using namespace std;

struct Student {
    int id;
    char name[50];
    float cgpa;
} s[10];

int countStudents = 0;

// Add a student (must be entered in sorted order by ID)
void addStudent() {
    if (countStudents >= 10) {
        cout << "Student limit reached!\n";
        return;
    }

    cout << "Enter Student ID (in ascending order): ";
    cin >> s[countStudents].id;
    cin.ignore();

    cout << "Enter Student Name: ";
    cin.getline(s[countStudents].name, 50);

    cout << "Enter CGPA: ";
    cin >> s[countStudents].cgpa;

    countStudents++;
    cout << "Student added successfully!\n";
}

// Display all students
void displayStudents() {
    if (countStudents == 0) {
        cout << "No students to display.\n";
        return;
    }

    cout << "\n--- Student Records ---\n";
    for (int i = 0; i < countStudents; i++) {
        cout << "ID: " << s[i].id
             << ", Name: " << s[i].name
             << ", CGPA: " << s[i].cgpa << "\n";
    }
}

// Binary Search by ID
void binarySearchByID() {
    if (countStudents == 0) {
        cout << "No students in database.\n";
        return;
    }

    int key;
    cout << "Enter ID to search: ";
    cin >> key;

    int low = 0, high = countStudents - 1;
    bool found = false;

    while (low <= high) {
        int mid = (low + high) / 2;

        if (s[mid].id == key) {
            cout << "\nStudent Found:\n";
            cout << "ID: " << s[mid].id
                 << ", Name: " << s[mid].name
                 << ", CGPA: " << s[mid].cgpa << "\n";
            found = true;
            break;
        }
        else if (s[mid].id < key)
            low = mid + 1;
        else
            high = mid - 1;
    }

    if (!found)
        cout << "Student ID not found.\n";
}

// Main menu
int main() {
    int choice;
    do {
        cout << "\n=== Student Database (Binary Search) ===\n";

C++ Program: Bubble Sort

#include <iostream>
#include <stdlib.h>
#include <conio.h>
using namespace std;

int main() {
    int n, i, j, arr[10], temp;

    cout << "Enter Total Number of Elements: ";
    cin >> n;
     
    cout << "Enter Elements: ";
    for (i = 0; i < n; i++) {
        cin >> arr[i];
    }

    // Bubble Sort Algorithm
    for (i = 0; i < n - 1; i++) {
        for (j = 0; j < n - i - 1; j++) {
            if (arr[j] > arr[j + 1]) {
                temp = arr[j];
                arr[j] = arr[j + 1];
                arr[j + 1] = temp;
            }
        }
    }

    cout << "Sorted Elements in Ascending Order: ";
    for (i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }

    return 0;
}

#include <stdlib.h>
#include <conio.h>

        cin >> arr[i];

    // Selection Sort Algorithm
    for (i = 0; i < n - 1; i++) {
        for (j = i + 1; j < n; j++) {
            if (arr[i] > arr[j]) {
                temp = arr[i];
                arr[i] = arr[j];
                arr[j] = temp;
            }
        }
    }

    cout << "Sorted Elements: ";
    for (i = 0; i < n; i++) {
        cout << arr[i] << " ";
    }

}
