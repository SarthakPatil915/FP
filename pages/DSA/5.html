#---------- Kruskalâ€™s Algorithm ----------
print("\n----- Kruskalâ€™s Algorithm -----\n")

# Graph as adjacency matrix
G = [
    [0, 4, 0, 0, 0, 2],
    [4, 0, 6, 0, 0, 3],
    [0, 6, 0, 3, 0, 1],
    [0, 0, 3, 0, 2, 0],
    [0, 0, 0, 2, 0, 4],
    [2, 3, 1, 0, 4, 0]
]

V = len(G)

# Step 1: Collect all edges (weight, start, end)
edges = []
for i in range(V):
    for j in range(i + 1, V):
        if G[i][j] != 0:
            edges.append((G[i][j], i, j))

# Step 2: Sort edges by weight
edges.sort()

# Step 3: Initialize parent array
parent = [i for i in range(V)]

# Function to find root of a vertex
def find(v):
    while parent[v] != v:
        v = parent[v]
    return v

# Function to join two sets
def union(u, v):
    root_u = find(u)
    root_v = find(v)
    if root_u != root_v:
        parent[root_v] = root_u

# Step 4: Kruskalâ€™s Algorithm
mst_edges = []
mst_weight = 0

print("Edge : Weight\n")
for weight, u, v in edges:
    if find(u) != find(v):
        union(u, v)
        mst_edges.append((u, v, weight))
        mst_weight += weight

# Step 5: Print result
for u, v, w in mst_edges:
    print(f"{u} - {v} : {w}")

print("\nTotal weight of Minimum Spanning Tree =", mst_weight)


#---------- Primâ€™s Algorithm ----------
print("\n----- Primâ€™s Algorithm -----\n")

INF = 9999999
V = 6

# Graph as adjacency matrix (same as above)
G = [
    [0, 4, 0, 0, 0, 2],
    [4, 0, 6, 0, 0, 3],
    [0, 6, 0, 3, 0, 1],
    [0, 0, 3, 0, 2, 0],
    [0, 0, 0, 2, 0, 4],
    [2, 3, 1, 0, 4, 0]
]

selected = [False] * V
no_edge = 0
selected[0] = True
total_weight = 0

print("Edge : Weight\n")

while no_edge < V - 1:
    minimum = INF
    x = 0
    y = 0

    for i in range(V):
        if selected[i]:
            for j in range(V):
                if (not selected[j]) and G[i][j]:
                    if minimum > G[i][j]:
                        minimum = G[i][j]
                        x = i
                        y = j

    print(f"{x} - {y} : {G[x][y]}")
    total_weight += G[x][y]
    selected[y] = True
    no_edge += 1

print("\nTotal weight of Minimum Spanning Tree =", total_weight)
