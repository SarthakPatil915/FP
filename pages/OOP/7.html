Java Program: Number Analyzer (Even, Odd, Prime, Palindrome)

import java.util.*; 

public class NumberAnalyzer { 
    public static boolean isEven(int num) { 
        return num % 2 == 0; 
    } 

    public static boolean isOdd(int num) { 
        return num % 2 != 0; 
    } 

    public static boolean isPrime(int num) { 
        if (num <= 1) return false; 
        for (int i = 2; i <= Math.sqrt(num); i++) { 
            if (num % i == 0) return false; 
        } 
        return true; 
    } 

    public static boolean isPalindrome(int num) { 
        int original = num, reversed = 0; 
        while (num != 0) { 
            reversed = reversed * 10 + num % 10; 
            num /= 10; 
        } 
        return original == reversed; 
    } 

    public static  int countMatching(List list, String type) { 
        int count = 0; 
        for (T item : list) { 
            int value = item.intValue(); 
            switch (type.toLowerCase()) { 
                case "even": 
                    if (isEven(value)) count++; 
                    break; 
                case "odd": 
                    if (isOdd(value)) count++; 
                    break; 
                case "prime": 
                    if (isPrime(value)) count++; 
                    break; 
                case "palindrome": 
                    if (isPalindrome(value)) count++; 
                    break; 
            } 
        } 
        return count; 
    } 

    public static void main(String[] args) { 
        Scanner sc = new Scanner(System.in); 
        List numbers = new ArrayList<>(); 

        System.out.print("Enter number of elements: "); 
        int n = sc.nextInt(); 

        System.out.println("Enter " + n + " integers:"); 
        for (int i = 0; i < n; i++) { 
            numbers.add(sc.nextInt()); 
        } 

        System.out.println("Even count: " + countMatching(numbers, "even")); 
        System.out.println("Odd count: " + countMatching(numbers, "odd")); 
        System.out.println("Prime count: " + countMatching(numbers, "prime")); 
        System.out.println("Palindrome count: " + countMatching(numbers, "palindrome")); 

        sc.close(); 
    } 
}
