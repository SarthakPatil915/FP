Java Program: Publication, Book & Magazine

class Publication { 
    protected String title; 
    protected double price; 
    protected int copies; 

    public void saleCopy() { 
        double total = price * copies; 
        System.out.println("Total sales for " + title + ": RS." + total); 
    } 
} 

class Book extends Publication { 
    private String author; 

    public void orderCopies(int numCopies) { 
        this.copies = numCopies; 
    } 

    public void setDetails(String title, double price, String author) { 
        this.title = title; 
        this.price = price; 
        this.author = author; 
    } 

    public void display() { 
        System.out.println("Book: " + title + " by " + author); 
        saleCopy(); 
    } 
} 

class Magazine extends Publication { 
    private int orderQty; 
    private String currentIssue; 

    public void receiveIssue(String issue, int qty) { 
        this.currentIssue = issue; 
        this.orderQty = qty; 
        this.copies = qty; 
    }  

    public void setDetails(String title, double price) { 
        this.title = title; 
        this.price = price; 
    } 

    public void display() { 
        System.out.println("Magazine: " + title + " - Issue: " + currentIssue + " - Ordered Copies: " + orderQty); 
        saleCopy(); 
    } 
} 

public class Main { 
    public static void main(String[] args) { 
        Book b1 = new Book(); 
        b1.setDetails("Java Programming", 450.0, "James Gosling"); 
        b1.orderCopies(10); 
        b1.display(); 

        Magazine m1 = new Magazine(); 
        m1.setDetails("Tech Monthly", 120.0); 
        m1.receiveIssue("September 2025", 25); 
        m1.display(); 
    } 
}
